### 4.7 为什么维度一定要先确认

Atlas 建 vector index 时必须知道 `numDimensions`。

这个值不能猜。

因为一旦索引维度和实际 embedding 维度不一致，后面你会遇到：

1. 插入失败
2. 检索失败
3. 检索结果为空

### 4.8 Atlas Vector Search 索引示例

向量集合建议叫：

1. `document_chunk_vectors`

文档形状建议是：

1. `userId`
2. `knowledgeBaseId`
3. `documentId`
4. `documentName`
5. `sequence`
6. `text`
7. `embedding`
8. `page`
9. `startIndex`
10. `endIndex`

Atlas Vector Search 索引配置示例：

```json
{
  "fields": [
    {
      "type": "vector",
      "path": "embedding",
      "numDimensions": 1536,
      "similarity": "cosine"
    },
    {
      "type": "filter",
      "path": "userId"
    },
    {
      "type": "filter",
      "path": "knowledgeBaseId"
    },
    {
      "type": "filter",
      "path": "documentId"
    }
  ]
}
```

### 4.9 为什么向量集合里的 id 推荐存字符串

这里推荐：

1. 业务表里继续用 `ObjectId`
2. 向量集合里把 `userId`、`knowledgeBaseId`、`documentId` 存成字符串

原因：

1. 向量集合不是业务事实表
2. Atlas 过滤条件直接用字符串更直观
3. 你在 raw collection 上做 `$vectorSearch` 时，不容易被 `ObjectId` 类型匹配坑住

### 5.6 `DocumentIndexingService` 推荐骨架

下面这段可以直接作为第一版骨架。

```ts
import { OpenAIEmbeddings } from '@langchain/openai';
import { RecursiveCharacterTextSplitter } from '@langchain/textsplitters';
import { BadRequestException, Injectable } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import { InjectConnection } from '@nestjs/mongoose';
import { Document as LangChainDocument } from 'langchain';
import { Collection, Db } from 'mongodb';
import { Connection } from 'mongoose';
import mammoth from 'mammoth';
import { PDFParse } from 'pdf-parse';
import { TEXT_UPLOAD_EXTENSIONS } from './documents.service';

type PrepareChunksInput = {
  userId: string;
  knowledgeBaseId: string;
  documentId: string;
  documentName: string;
  extension: string;
  content?: string;
  file?: Express.Multer.File;
};

export type PreparedChunk = {
  sequence: number;
  content: string;
  page?: number;
  startIndex?: number;
  endIndex?: number;
};

@Injectable()
export class DocumentIndexingService {
  constructor(
    private readonly configService: ConfigService,
    @InjectConnection() private readonly connection: Connection,
  ) {}

  private get db(): Db {
    if (!this.connection.db) {
      throw new Error('Mongo connection is not ready');
    }

    return this.connection.db;
  }

  private getVectorCollectionName(): string {
    return (
      this.configService.get<string>('MONGODB_VECTOR_COLLECTION') ||
      'document_chunk_vectors'
    );
  }

  private getVectorIndexName(): string {
    return (
      this.configService.get<string>('MONGODB_VECTOR_INDEX') ||
      'document_chunk_vector_index'
    );
  }

  private getVectorCollection(): Collection {
    return this.db.collection(this.getVectorCollectionName());
  }

  private createEmbeddings() {
    return new OpenAIEmbeddings({
      model: this.configService.get<string>('OPENAI_EMBEDDING_MODEL'),
      apiKey: this.configService.get<string>('OPENAI_EMBEDDING_API_KEY'),
      batchSize:
        Math.min(
          this.configService.get<number>('OPENAI_EMBEDDING_BATCH_SIZE') ?? 10,
          10,
        ) || 10,
      configuration: {
        baseURL: this.configService.get<string>('OPENAI_EMBEDDING_BASE_URL'),
      },
    });
  }

  private async loadSourceDocuments(
    params: PrepareChunksInput,
  ): Promise<LangChainDocument[]> {
    if (TEXT_UPLOAD_EXTENSIONS.has(params.extension)) {
      return [
        new LangChainDocument({
          pageContent: params.content ?? '',
          metadata: {
            documentId: params.documentId,
            documentName: params.documentName,
          },
        }),
      ];
    }

    if (params.extension === 'pdf') {
      if (!params.file) {
        throw new BadRequestException('pdf 文件缺少原始文件内容');
      }

      const parser = new PDFParse({
        data: params.file.buffer,
      });

      try {
        const result = await parser.getText();

        return result.pages
          .filter((page) => page.text.trim().length > 0)
          .map(
            (page) =>
              new LangChainDocument({
                pageContent: page.text,
                metadata: {
                  documentId: params.documentId,
                  documentName: params.documentName,
                  page: page.num,
                },
              }),
          );
      } finally {
        await parser.destroy();
      }
    }

    if (params.extension === 'docx') {
      if (!params.file) {
        throw new BadRequestException('docx 文件缺少原始文件内容');
      }

      const result = await mammoth.extractRawText({
        buffer: params.file.buffer,
      });

      return [
        new LangChainDocument({
          pageContent: result.value,
          metadata: {
            documentId: params.documentId,
            documentName: params.documentName,
          },
        }),
      ];
    }

    if (params.extension === 'doc') {
      throw new BadRequestException('当前仅支持解析 docx，请先转换为 docx');
    }

    throw new BadRequestException(`不支持的扩展名: ${params.extension}`);
  }

  private async splitDocuments(documents: LangChainDocument[]) {
    const splitter = new RecursiveCharacterTextSplitter({
      chunkSize: 1000,
      chunkOverlap: 200,
    });

    return splitter.splitDocuments(documents);
  }

  async prepareChunks(params: PrepareChunksInput): Promise<PreparedChunk[]> {
    const sourceDocuments = await this.loadSourceDocuments(params);
    const splitDocuments = await this.splitDocuments(sourceDocuments);

    return splitDocuments
      .map((document, index) => {
        const page =
          typeof document.metadata.page === 'number'
            ? document.metadata.page
            : typeof document.metadata.loc?.pageNumber === 'number'
              ? document.metadata.loc.pageNumber
              : undefined;

        const startIndex =
          typeof document.metadata.startIndex === 'number'
            ? document.metadata.startIndex
            : undefined;

        const content = document.pageContent.trim();

        return {
          sequence: index,
          content,
          page,
          startIndex,
          endIndex:
            typeof startIndex === 'number'
              ? startIndex + content.length
              : undefined,
        };
      })
      .filter((chunk) => chunk.content.length > 0);
  }

  async replaceDocumentVectors(params: {
    userId: string;
    knowledgeBaseId: string;
    documentId: string;
    documentName: string;
    chunks: PreparedChunk[];
  }): Promise<void> {
    const collection = this.getVectorCollection();

    await collection.deleteMany({
      userId: params.userId,
      documentId: params.documentId,
    });

    if (!params.chunks.length) {
      return;
    }

    const embeddings = await this.createEmbeddings().embedDocuments(
      params.chunks.map((chunk) => chunk.content),
    );

    await collection.insertMany(
      params.chunks.map((chunk, index) => ({
        userId: params.userId,
        knowledgeBaseId: params.knowledgeBaseId,
        documentId: params.documentId,
        documentName: params.documentName,
        sequence: chunk.sequence,
        text: chunk.content,
        embedding: embeddings[index],
        page: chunk.page,
        startIndex: chunk.startIndex,
        endIndex: chunk.endIndex,
      })),
    );
  }

  async deleteDocumentVectors(userId: string, documentId: string): Promise<void> {
    await this.getVectorCollection().deleteMany({
      userId,
      documentId,
    });
  }

  async semanticSearch(params: {
    userId: string;
    knowledgeBaseId: string;
    question: string;
    topK: number;
  }) {
    const queryVector = await this.createEmbeddings().embedQuery(
      params.question,
    );

    return this.getVectorCollection()
      .aggregate([
        {
          $vectorSearch: {
            index: this.getVectorIndexName(),
            path: 'embedding',
            queryVector,
            numCandidates: Math.max(params.topK * 10, 50),
            limit: params.topK,
            filter: {
              userId: params.userId,
              knowledgeBaseId: params.knowledgeBaseId,
            },
          },
        },
        {
          $project: {
            _id: 0,
            documentId: 1,
            documentName: 1,
            sequence: 1,
            text: 1,
            page: 1,
            startIndex: 1,
            endIndex: 1,
            score: {
              $meta: 'vectorSearchScore',
            },
          },
        },
      ])
      .toArray();
  }
}
```

注意：

1. 当前仓库里锁定的 `@langchain/textsplitters@1.0.1` 不支持 `addStartIndex`
2. 所以第一版先允许 `startIndex / endIndex` 为空
3. 如果后面你确实需要精确字符位置，再单独补一层手动计算

### 5.7 上面这个 service 的思维顺序

你要按下面这个固定顺序理解：

```text
拿正文
  -> 切 chunk
  -> chunk 统一成结构
  -> 生成 embedding
  -> 写入向量集合
  -> 后面再做检索
```

### 5.8 为什么这里不用 `MongoClient`

当前项目已经通过 `MongooseModule.forRootAsync` 建好了连接。

所以更适合：

1. 用 `@InjectConnection()`
2. 拿到底层 `connection.db`
3. 操作 raw collection

这样比在 service 里再 `new MongoClient(...)` 更稳。

### 5.9 为什么 `prepareChunks` 和 `replaceDocumentVectors` 分开

因为 `document_chunks` 是业务表，`document_chunk_vectors` 是向量表。

更合理的职责分工是：

1. `DocumentIndexingService.prepareChunks()` 只负责得到 chunk 结果
2. `DocumentsService.replaceDocumentChunks()` 负责写业务表
3. `DocumentIndexingService.replaceDocumentVectors()` 负责写向量表

这样不会把业务表写入和向量表写入搅在一起。

### 6.6 `syncDocumentIndex` 推荐实现

```ts
private async syncDocumentIndex(params: {
  userId: string;
  knowledgeBaseId: string;
  documentId: string;
  documentName: string;
  extension: string;
  content?: string;
  file?: Express.Multer.File;
}) {
  const chunks = await this.documentIndexingService.prepareChunks(params);

  await this.replaceDocumentChunks(
    params.userId,
    params.knowledgeBaseId,
    params.documentId,
    chunks,
  );

  await this.documentIndexingService.replaceDocumentVectors({
    userId: params.userId,
    knowledgeBaseId: params.knowledgeBaseId,
    documentId: params.documentId,
    documentName: params.documentName,
    chunks,
  });
}
```

### 6.7 这个方法的思维逻辑

顺序一定是：

```text
先 prepareChunks
  -> 再写 document_chunks
  -> 再写向量集合
```

为什么不是先写向量？

因为 `document_chunks` 是你的业务事实。

即使向量检索后面要重建，`document_chunks` 也应该先可见、可查、可调试。

```ts
}
```

然后在 `remove()` 里替换原来的 chunk 删除逻辑：

```ts
await this.removeDocumentIndex(userId, id);
```

### 6.11 为什么删除时也要单独抽方法

因为删除文档索引本质上是一件完整的业务动作：

1. 删业务 chunk
2. 删向量数据

抽成一个方法后，后面如果你要做“重建索引前先清理旧索引”，也可以直接复用。

### 6.12 当前先不要急着做 `/documents/:id/reindex`

原因很现实。

当前 `StorageService` 只有：

1. `uploadFile`
2. `deleteFile`
3. `getSignedUrl`

它还没有“下载原文件 buffer”的能力。

这意味着：

1. 编辑器文档和 `md/txt` 可以直接从 `content` 重建
2. `pdf/docx` 如果要做重建索引，你还得先补下载逻辑

所以第一版先别扩 API，先把“创建即索引、删除即清理”做好。

### 8.4 `ChatService` 推荐骨架

```ts
import { Injectable, NotFoundException } from '@nestjs/common';
import { InjectModel } from '@nestjs/mongoose';
import { ChatOpenAI } from '@langchain/openai';
import { HumanMessage, SystemMessage } from 'langchain';
import { ConfigService } from '@nestjs/config';
import { Model } from 'mongoose';
import { toObjectId } from 'src/common/utils/object-id.util';
import { DocumentIndexingService } from 'src/documents/document-indexing.service';
import {
  KnowledgeBase,
  KnowledgeBaseDocument,
} from 'src/knowledge_bases/schemas/knowledge_base.schema';
import {
  ChatMessage,
  ChatMessageDocument,
  ChatMessageType,
} from './schemas/chat_message.schema';
import {
  ChatSession,
  ChatSessionDocument,
} from './schemas/chat_session.schema';
import { AskChatDto } from './dto/ask-chat.dto';

@Injectable()
export class ChatService {
  constructor(
    @InjectModel(KnowledgeBase.name)
    private readonly knowledgeBaseModel: Model<KnowledgeBaseDocument>,
    @InjectModel(ChatSession.name)
    private readonly chatSessionModel: Model<ChatSessionDocument>,
    @InjectModel(ChatMessage.name)
    private readonly chatMessageModel: Model<ChatMessageDocument>,
    private readonly configService: ConfigService,
    private readonly documentIndexingService: DocumentIndexingService,
  ) {}

  private createChatModel() {
    return new ChatOpenAI({
      model: this.configService.get<string>('OPENAI_CHAT_MODEL'),
      apiKey: this.configService.get<string>('OPENAI_CHAT_API_KEY'),
      configuration: {
        baseURL: this.configService.get<string>('OPENAI_CHAT_BASE_URL'),
      },
    });
  }

  private async ensureKnowledgeBaseAccess(
    userId: string,
    knowledgeBaseId: string,
  ) {
    const knowledgeBase = await this.knowledgeBaseModel
      .findOne({
        _id: toObjectId(knowledgeBaseId),
        userId: toObjectId(userId),
      })
      .select({ _id: 1 })
      .lean()
      .exec();

    if (!knowledgeBase) {
      throw new NotFoundException('知识库不存在');
    }
  }

  private buildSessionTitle(question: string): string {
    return question.trim().slice(0, 50);
  }

  private async findOrCreateSession(
    userId: string,
    knowledgeBaseId: string,
    sessionId: string | undefined,
    question: string,
  ) {
    if (sessionId) {
      const session = await this.chatSessionModel
        .findOne({
          _id: toObjectId(sessionId),
          userId: toObjectId(userId),
          knowledgeBaseId: toObjectId(knowledgeBaseId),
        })
        .exec();

      if (!session) {
        throw new NotFoundException('会话不存在');
      }

      return session;
    }

    const session = new this.chatSessionModel({
      userId: toObjectId(userId),
      knowledgeBaseId: toObjectId(knowledgeBaseId),
      title: this.buildSessionTitle(question),
    });

    await session.save();
    return session;
  }

  private async getNextMessageSequence(sessionId: string): Promise<number> {
    const latest = await this.chatMessageModel
      .findOne({
        sessionId: toObjectId(sessionId),
      })
      .sort({ sequence: -1 })
      .select({ sequence: 1 })
      .lean()
      .exec();

    return latest ? latest.sequence + 1 : 0;
  }

  private async createMessage(params: {
    userId: string;
    knowledgeBaseId: string;
    sessionId: string;
    messageType: ChatMessageType;
    content: string;
    sources?: Array<{
      documentId: string;
      documentName: string;
      chunkSequence: number;
      page?: number;
      startIndex?: number;
      endIndex?: number;
    }>;
  }) {
    const sequence = await this.getNextMessageSequence(params.sessionId);

    const message = new this.chatMessageModel({
      userId: toObjectId(params.userId),
      knowledgeBaseId: toObjectId(params.knowledgeBaseId),
      sessionId: toObjectId(params.sessionId),
      messageType: params.messageType,
      content: params.content,
      sequence,
      sources: params.sources,
    });

    await message.save();
    return message;
  }

  private buildContextText(
    hits: Array<{
      documentName: string;
      sequence: number;
      text: string;
      page?: number;
    }>,
  ): string {
    return hits
      .map((hit, index) => {
        const location =
          typeof hit.page === 'number'
            ? `page=${hit.page}, sequence=${hit.sequence}`
            : `sequence=${hit.sequence}`;

        return [
          `[片段 ${index + 1}]`,
          `文档: ${hit.documentName}`,
          `位置: ${location}`,
          hit.text,
        ].join('\n');
      })
      .join('\n\n');
  }

  private normalizeModelOutput(content: unknown): string {
    if (typeof content === 'string') {
      return content.trim();
    }

    if (Array.isArray(content)) {
      return content
        .map((item) => {
          if (typeof item === 'string') {
            return item;
          }

          if (
            item &&
            typeof item === 'object' &&
            'text' in item &&
            typeof item.text === 'string'
          ) {
            return item.text;
          }

          return '';
        })
        .join('\n')
        .trim();
    }

    return '';
  }

  async ask(userId: string, dto: AskChatDto) {
    await this.ensureKnowledgeBaseAccess(userId, dto.knowledgeBaseId);

    const session = await this.findOrCreateSession(
      userId,
      dto.knowledgeBaseId,
      dto.sessionId,
      dto.question,
    );

    await this.createMessage({
      userId,
      knowledgeBaseId: dto.knowledgeBaseId,
      sessionId: session.id,
      messageType: ChatMessageType.Human,
      content: dto.question,
    });

    const hits = await this.documentIndexingService.semanticSearch({
      userId,
      knowledgeBaseId: dto.knowledgeBaseId,
      question: dto.question,
      topK: dto.topK ?? 5,
    });

    const contextText = this.buildContextText(hits);

    const model = this.createChatModel();
    const response = await model.invoke([
      new SystemMessage(
        '你是知识库问答助手。只能基于提供的上下文回答；如果上下文不足，直接说明不知道，不要编造。',
      ),
      new HumanMessage(
        [
          `用户问题：${dto.question}`,
          '',
          '可用上下文：',
          contextText || '无可用上下文',
        ].join('\n'),
      ),
    ]);

    const answer = this.normalizeModelOutput(response.content);

    const sources = hits.map((hit) => ({
      documentId: hit.documentId,
      documentName: hit.documentName,
      chunkSequence: hit.sequence,
      page: hit.page,
      startIndex: hit.startIndex,
      endIndex: hit.endIndex,
    }));

    await this.createMessage({
      userId,
      knowledgeBaseId: dto.knowledgeBaseId,
      sessionId: session.id,
      messageType: ChatMessageType.Ai,
      content: answer,
      sources,
    });

    return {
      sessionId: session.id,
      answer,
      sources,
    };
  }
}
```
